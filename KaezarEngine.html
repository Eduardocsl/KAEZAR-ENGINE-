# Análise: Exportação para GitHub

Analisando o código-fonte fornecido, especificamente a função de exportação para GitHub:

## Problema Identificado

NÃO, os patches NÃO estão sendo enviados para o GitHub ao exportar.

Encontrei a função `exportToGitHub()` no código (linha ~1850-1920). Ela está fazendo o seguinte:

1. ✅ Coleta o código HTML completo
2. ✅ Cria um blob no GitHub
3. ✅ Cria uma tree
4. ✅ Cria um commit
5. ✅ Atualiza a referência do branch

MAS:
- ❌ Não inclui os patches do AI Agent
- ❌ Não inclui o histórico de conversas
- ❌ Não cria arquivos separados para patches
- ❌ Apenas exporta o HTML principal

## Solução: Exportar HTML + Patches

async function exportToGitHub() {
  const token = localStorage.getItem('github_token');
  const repo = localStorage.getItem('github_repo');
  const branch = localStorage.getItem('github_branch') || 'main';
  
  if (!token || !repo) {
    logConsole('error', '❌ Configure GitHub Token e Repo primeiro!');
    return;
  }
  
  const [owner, repoName] = repo.split('/');
  if (!owner || !repoName) {
    logConsole('error', '❌ Formato do repo inválido. Use: usuario/repositorio');
    return;
  }
  
  try {
    logConsole('info', '📤 Exportando para GitHub...');
    
    // 1. Pegar referência do branch
    const refRes = await fetch(`https://api.github.com/repos/${owner}/${repoName}/git/ref/heads/${branch}`, {
      headers: { 'Authorization': `token ${token}` }
    });
    
    if (!refRes.ok) {
      throw new Error(`Branch ${branch} não encontrado`);
    }
    
    const refData = await refRes.json();
    const latestCommitSha = refData.object.sha;
    
    // 2. Pegar tree do último commit
    const commitRes = await fetch(`https://api.github.com/repos/${owner}/${repoName}/git/commits/${latestCommitSha}`, {
      headers: { 'Authorization': `token ${token}` }
    });
    const commitData = await commitRes.json();
    const baseTreeSha = commitData.tree.sha;
    
    // 3. Preparar arquivos para commit
    const files = [];
    
    // 3a. HTML principal
    const htmlContent = document.documentElement.outerHTML;
    const htmlBlob = await fetch(`https://api.github.com/repos/${owner}/${repoName}/git/blobs`, {
      method: 'POST',
      headers: { 'Authorization': `token ${token}`, 'Content-Type': 'application/json' },
      body: JSON.stringify({
        content: btoa(unescape(encodeURIComponent(htmlContent))),
        encoding: 'base64'
      })
    }).then(r => r.json());
    
    files.push({
      path: 'index.html',
      mode: '100644',
      type: 'blob',
      sha: htmlBlob.sha
    });
    
    // 3b. Patches do AI Agent (se existirem)
    const aiMessages = JSON.parse(localStorage.getItem('ai_messages') || '[]');
    if (aiMessages.length > 0) {
      const patchesContent = aiMessages
        .filter(msg => msg.role === 'assistant' && msg.content.includes('
'))
.map((msg, idx) => {
const timestamp = new Date(msg.timestamp || Date.now()).toISOString();
return `## Patch ${idx + 1} - ${timestamp}\n\n${msg.content}\n\n---\n`;
})
.join('\n');

if (patchesContent.trim()) {
const patchBlob = await fetch(`https://api.github.com/repos/${owner}/${repoName}/git/blobs`, {
method: 'POST',
headers: { 'Authorization': `token ${token}`, 'Content-Type': 'application/json' },
body: JSON.stringify({
content: btoa(unescape(encodeURIComponent(patchesContent))),
encoding: 'base64'
})
}).then(r => r.json());

files.push({
path: 'AI_PATCHES.md',
mode: '100644',
type: 'blob',
sha: patchBlob.sha
});
}
}

// 3c. Histórico completo de conversas
if (aiMessages.length > 0) {
const historyContent = JSON.stringify(aiMessages, null, 2);
const historyBlob = await fetch(`https://api.github.com/repos/${owner}/${repoName}/git/blobs`, {
method: 'POST',
headers: { 'Authorization': `token ${token}`, 'Content-Type': 'application/json' },
body: JSON.stringify({
content: btoa(unescape(encodeURIComponent(historyContent))),
encoding: 'base64'
})
}).then(r => r.json());

files.push({
path: 'ai_history.json',
mode: '100644',
type: 'blob',
sha: historyBlob.sha
});
}

// 4. Criar nova tree
const treeRes = await fetch(`https://api.github.com/repos/${owner}/${repoName}/git/trees`, {
method: 'POST',
headers: { 'Authorization': `token ${token}`, 'Content-Type': 'application/json' },
body: JSON.stringify({
base_tree: baseTreeSha,
tree: files
})
});

if (!treeRes.ok) {
throw new Error('Erro ao criar tree');
}

const treeData = await treeRes.json();

// 5. Criar commit
const patchCount = files.length - 1; // -1 porque index.html sempre existe
const commitMessage = patchCount > 0
? `🚀 KaezarEngine v2 update + ${patchCount} AI patch(es)`
: '🚀 KaezarEngine v2 update';

const newCommitRes = await fetch(`https://api.github.com/repos/${owner}/${repoName}/git/commits`, {
method: 'POST',
headers: { 'Authorization': `token ${token}`, 'Content-Type': 'application/json' },
body: JSON.stringify({
message: commitMessage,
tree: treeData.sha,
parents: [latestCommitSha]
})
});

if (!newCommitRes.ok) {
throw new Error('Erro ao criar commit');
}

const newCommitData = await newCommitRes.json();

// 6. Atualizar referência do branch
const updateRefRes = await fetch(`https://api.github.com/repos/${owner}/${repoName}/git/refs/heads/${branch}`, {
method: 'PATCH',
headers: { 'Authorization': `token ${token}`, 'Content-Type': 'application/json' },
body: JSON.stringify({ sha: newCommitData.sha })
});

if (!updateRefRes.ok) {
throw new Error('Erro ao atualizar branch');
}

logConsole('info', `✅ Exportado para GitHub! ${files.length} arquivo(s) commitado(s).`);
logConsole('info', `📦 Commit: ${newCommitData.sha.substring(0, 7)}`);

if (patchCount > 0) {
logConsole('info', `🤖 ${patchCount} patch(es) AI incluído(s) em AI_PATCHES.md`);
}

} catch (err) {
logConsole('error', `❌ Erro ao exportar: ${err.message}`);
console.error(err);
}
}

// Substituir a função applyAIPatch() completamente
function applyAIPatch(code, patchIndex = 0) {
try {
// 1. Limpar código de markdown
let cleanCode = code.trim();

// Remover blocos de código markdown
const codeBlockRegex = /
(?:javascript|js|html|css)?\s*\n([\s\S]*?)\n
/g;
const matches = [...cleanCode.matchAll(codeBlockRegex)];

if (matches.length > 0) {
// Se tem blocos de código, extrair apenas o conteúdo
cleanCode = matches.map(m => m[1]).join('\n\n');
}

// Remover comentários de markdown (linhas começando com #, -, *, etc)
cleanCode = cleanCode
.split('\n')
.filter(line => {
const trimmed = line.trim();
return !(
trimmed.startsWith('#') ||
trimmed.startsWith('**') ||
trimmed.startsWith('##') ||
trimmed.startsWith('---') ||
trimmed.startsWith('
trimmed.startsWith('REQUER_RESTART:')
);
})
.join('\n')
.trim();

if (!cleanCode) {
logConsole('warn', `🔬 Patch ${patchIndex + 1}: Nenhum código válido encontrado`);
return false;
}

// 2. Detectar tipo de código
const isCSS = cleanCode.includes('{') && cleanCode.includes('}') &&
(cleanCode.includes(':root') || cleanCode.includes('color:') || cleanCode.includes('background:'));
const isHTML = cleanCode.trim().startsWith('<') && cleanCode.includes('>');

// 3. Aplicar baseado no tipo
if (isCSS) {
// Aplicar CSS
let styleTag = document.getElementById('ai-injected-styles');
if (!styleTag) {
styleTag = document.createElement('style');
styleTag.id = 'ai-injected-styles';
document.head.appendChild(styleTag);
}
styleTag.textContent += '\n' + cleanCode;
logConsole('info', `🔬 Patch ${patchIndex + 1}: CSS aplicado (${cleanCode.length} chars)`);
return true;

} else if (isHTML) {
// HTML não pode ser aplicado em runtime
logConsole('warn', `🔬 Patch ${patchIndex + 1}: HTML detectado - requer restart`);
return false;

} else {
// Tentar como JavaScript
// 3a. Validar sintaxe
try {
new Function(cleanCode);
} catch (syntaxError) {
logConsole('error', `🔬 Patch ${patchIndex + 1}: Sintaxe inválida - ${syntaxError.message}`);
return false;
}

// 3b. Executar em contexto isolado
try {
const result = (function() {
return eval(cleanCode);
}).call(window);

logConsole('info', `🔬 Patch ${patchIndex + 1}: JavaScript aplicado (${cleanCode.length} chars)`);

// Se retornou uma função, tentar registrá-la globalmente
if (typeof result === 'function') {
const funcName = result.name || `aiPatch${patchIndex}`;
window[funcName] = result;
logConsole('info', `🔬 Função ${funcName}() registrada globalmente`);
}

return true;

} catch (runtimeError) {
logConsole('error', `🔬 Patch ${patchIndex + 1}: Erro runtime - ${runtimeError.message}`);
console.error('Código que falhou:', cleanCode);
return false;
}
}

} catch (err) {
logConsole('error', `🔬 Patch ${patchIndex + 1}: Erro inesperado - ${err.message}`);
console.error(err);
return false;
}
}

function extractAndApplyPatches(aiReply) {
// Extrair blocos de código
const codeBlockRegex = /
(?:javascript|js|html|css)?\s*\n([\s\S]*?)\n
/g;
const codeBlocks = [...aiReply.matchAll(codeBlockRegex)];

if (codeBlocks.length === 0) {
logConsole('info', '🔬 Nenhum patch de código encontrado na resposta');
return { applied: 0, failed: 0, requiresRestart: false };
}

let appliedCount = 0;
let failedCount = 0;
let requiresRestart = false;

// Verificar se requer restart
if (aiReply.includes('') || aiReply.includes('REQUER_RESTART: true')) {
requiresRestart = true;
}

// Aplicar cada patch
codeBlocks.forEach((match, idx) => {
const code = match[1];
const success = applyAIPatch(code, idx);

if (success) {
appliedCount++;
} else {
failedCount++;
}
});

// Log resumo
if (appliedCount > 0) {
logConsole('info', `✅ ${appliedCount}/${codeBlocks.length} patch(es) aplicado(s) com sucesso`);
}

if (failedCount > 0) {
logConsole('warn', `⚠️ ${failedCount}/${codeBlocks.length} patch(es) falharam`);
}

if (requiresRestart) {
logConsole('warn', '🔄 Alguns patches requerem RESTART do app para funcionar');

// Mostrar botão de restart
setTimeout(() => {
if (confirm('⚠️ Patches aplicados requerem reiniciar o app.\n\nDeseja recarregar agora?')) {
location.reload();
}
}, 1000);
}

return { applied: appliedCount, failed: failedCount, requiresRestart };
}

// Adicionar sistema de gerenciamento de patches aplicados
const PatchManager = {
  // Estrutura: { id, code, type, applied, timestamp, description }
  patches: [],
  
  init() {
    const saved = localStorage.getItem('applied_patches');
    this.patches = saved ? JSON.parse(saved) : [];
    this.reapplyAll();
  },
  
  add(code, type, description = '') {
    const patch = {
      id: Date.now().toString(36) + Math.random().toString(36).substr(2),
      code: code.trim(),
      type: type, // 'js', 'css', 'html'
      applied: false,
      timestamp: Date.now(),
      description: description,
      hash: this.hashCode(code)
    };
    
    // Verificar duplicatas
    const exists = this.patches.find(p => p.hash === patch.hash);
    if (exists) {
      logConsole('warn', '⚠️ Patch duplicado detectado, ignorando');
      return exists.id;
    }
    
    this.patches.push(patch);
    this.save();
    return patch.id;
  },
  
  apply(patchId) {
    const patch = this.patches.find(p => p.id === patchId);
    if (!patch) return false;
    
    if (patch.applied) {
      logConsole('warn', `⚠️ Patch ${patchId} já está aplicado`);
      return true;
    }
    
    const success = this.executePatch(patch);
    if (success) {
      patch.applied = true;
      patch.appliedAt = Date.now();
      this.save();
    }
    
    return success;
  },
  
  unapply(patchId) {
    const patch = this.patches.find(p => p.id === patchId);
    if (!patch || !patch.applied) return false;
    
    // CSS pode ser removido
    if (patch.type === 'css') {
      const styleTag = document.getElementById(`patch-${patchId}`);
      if (styleTag) {
        styleTag.remove();
        patch.applied = false;
        this.save();
        return true;
      }
    }
    
    // JS não pode ser "desaplicado" sem reload
    logConsole('warn', '⚠️ Patches JS requerem reload para remover');
    return false;
  },
  
  remove(patchId) {
    const index = this.patches.findIndex(p => p.id === patchId);
    if (index === -1) return false;
    
    const patch = this.patches[index];
    if (patch.applied) {
      this.unapply(patchId);
    }
    
    this.patches.splice(index, 1);
    this.save();
    return true;
  },
  
  executePatch(patch) {
    try {
      if (patch.type === 'css') {
        let styleTag = document.getElementById(`patch-${patch.id}`);
        if (!styleTag) {
          styleTag = document.createElement('style');
          styleTag.id = `patch-${patch.id}`;
          styleTag.setAttribute('data-patch-desc', patch.description);
          document.head.appendChild(styleTag);
        }
        styleTag.textContent = patch.code;
        return true;
        
      } else if (patch.type === 'js') {
        // Validar sintaxe primeiro
        new Function(patch.code);
        
        // Executar
        (function() {
          return eval(patch.code);
        }).call(window);
        
        return true;
        
      } else if (patch.type === 'html') {
        logConsole('warn', '⚠️ Patches HTML requerem restart');
        return false;
      }
      
    } catch (err) {
      logConsole('error', `❌ Erro ao aplicar patch ${patch.id}: ${err.message}`);
      return false;
    }
  },
  
  reapplyAll() {
    const applied = this.patches.filter(p => p.applied);
    let successCount = 0;
    
    applied.forEach(patch => {
      patch.applied = false; // Reset para reaplicar
      if (this.executePatch(patch)) {
        patch.applied = true;
        successCount++;
      }
    });
    
    if (successCount > 0) {
      logConsole('info', `🔄 ${successCount} patch(es) reaplicado(s) após reload`);
    }
  },
  
  save() {
    localStorage.setItem('applied_patches', JSON.stringify(this.patches));
  },
  
  hashCode(str) {
    let hash = 0;
    for (let i = 0; i < str.length; i++) {
      const char = str.charCodeAt(i);
      hash = ((hash << 5) - hash) + char;
      hash = hash & hash;
    }
    return hash.toString(36);
  },
  
  getStats() {
    return {
      total: this.patches.length,
      applied: this.patches.filter(p => p.applied).length,
      js: this.patches.filter(p => p.type === 'js').length,
      css: this.patches.filter(p => p.type === 'css').length,
      html: this.patches.filter(p => p.type === 'html').length
    };
  },
  
  exportAll() {
    return {
      version: '2.0',
      exported: new Date().toISOString(),
      patches: this.patches
    };
  },
  
  importPatches(data) {
    if (!data.patches || !Array.isArray(data.patches)) {
      throw new Error('Formato de importação inválido');
    }
    
    let imported = 0;
    data.patches.forEach(patch => {
      const exists = this.patches.find(p => p.hash === patch.hash);
      if (!exists) {
        this.patches.push(patch);
        imported++;
      }
    });
    
    this.save();
    logConsole('info', `📥 ${imported} patch(es) importado(s)`);
    return imported;
  }
};

  PatchManager.init();
